function verifyCert(){ const certId = document.getElementById("certInput").value.trim(); const result = document.getElementById("result"); /* CHECK EMPTY */ if(!certId){ result.innerHTML = "
Enter a certificate ID.
"; return; } /* VALIDATE FORMAT */ const certPattern = /^AMS-[EDSB]-\d{8}$/; if(!certPattern.test(certId)){ result.innerHTML = `
Invalid certificate format.
`; return; } /* CHECK CERTIFICATE REGISTRY */ fetch("../data/certificates.json") .then(res => res.json()) .then(certificates => { const cert = certificates.find(c => c.id === certId); if(cert){ let moduleType = "AMS Training Module"; if(cert.id.includes("-E-")) moduleType = "FMCSA Employee Awareness"; if(cert.id.includes("-D-")) moduleType = "FMCSA DER Training"; if(cert.id.includes("-S-")) moduleType = "FMCSA Supervisor Training"; if(cert.id.includes("-B-")) moduleType = "FMCSA Drug & Alcohol Training"; const dateObj = new Date(cert.date); const formattedDate = dateObj.toLocaleDateString("en-US",{ month:"2-digit", day:"2-digit", year:"numeric" }); result.innerHTML = `
Certificate Verified ✔

Name: ${cert.name}
Course: ${moduleType}
Date Completed: ${formattedDate}
Certificate ID: ${cert.id}
`; }else{ result.innerHTML = `
Certificate NOT FOUND
`; } }) .catch(()=>{ result.innerHTML = `
Error loading certificate registry
`; }); }
